-- 1. TABLA DE RECETAS (NUTRICIÓN)
CREATE TABLE IF NOT EXISTS gym_recipes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    ingredients TEXT,
    instructions TEXT,
    image_url TEXT,
    category TEXT,
    calories INTEGER,
    protein INTEGER,
    carbs INTEGER,
    fats INTEGER,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Asegurar columnas Ultra Pro en Recetas
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS prep_time TEXT DEFAULT '20 min';
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS difficulty TEXT DEFAULT 'Media';
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS diet_type TEXT DEFAULT 'Equilibrada';

-- 2. TABLA DE RUTINAS
CREATE TABLE IF NOT EXISTS gym_routines (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    exercises JSONB DEFAULT '[]'::jsonb,
    level TEXT,
    goal TEXT,
    image_url TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Asegurar columnas Ultra Pro en Rutinas
ALTER TABLE gym_routines ADD COLUMN IF NOT EXISTS target_gender TEXT DEFAULT 'Unisex';
ALTER TABLE gym_routines ADD COLUMN IF NOT EXISTS target_age_group TEXT DEFAULT 'Adulto';
ALTER TABLE gym_routines ADD COLUMN IF NOT EXISTS health_conditions TEXT DEFAULT 'Ninguna';
ALTER TABLE gym_routines ADD COLUMN IF NOT EXISTS plan_duration_weeks INTEGER DEFAULT 4;
ALTER TABLE gym_routines ADD COLUMN IF NOT EXISTS difficulty_level TEXT DEFAULT 'Intermedio';

-- 3. TABLA DE ENTRENADORES
CREATE TABLE IF NOT EXISTS gym_trainers (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    specialty TEXT,
    bio TEXT,
    image_url TEXT,
    contact_info TEXT,
    rating DECIMAL(3,2) DEFAULT 5.0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- ######################################################
-- # POLÍTICAS DE SEGURIDAD (RLS)                       #
-- ######################################################
-- IMPORTANTE: Estas políticas permiten acceso total (CRUD) para facilitar el desarrollo inicial.

ALTER TABLE gym_recipes ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow all for gym_recipes" ON gym_recipes;
CREATE POLICY "Allow all for gym_recipes" ON gym_recipes FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE gym_routines ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow all for gym_routines" ON gym_routines;
CREATE POLICY "Allow all for gym_routines" ON gym_routines FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE gym_trainers ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow all for gym_trainers" ON gym_trainers;
CREATE POLICY "Allow all for gym_trainers" ON gym_trainers FOR ALL USING (true) WITH CHECK (true);
