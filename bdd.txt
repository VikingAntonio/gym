-- Tabla de Recetas
CREATE TABLE gym_recipes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    ingredients TEXT,
    instructions TEXT,
    image_url TEXT,
    category TEXT, -- Desayuno, Almuerzo, Cena, Snack
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Tabla de Rutinas
CREATE TABLE gym_routines (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    exercises TEXT, -- Lista de ejercicios en formato texto o JSON
    level TEXT, -- Principiante, Intermedio, Avanzado
    goal TEXT, -- Perder peso, Ganar músculo, Mantenerse
    image_url TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Tabla de Entrenadores
CREATE TABLE gym_trainers (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    specialty TEXT,
    bio TEXT,
    image_url TEXT,
    contact_info TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Habilitar RLS (Row Level Security) - Para propósitos de este ejercicio, permitiremos lectura y escritura anónima si es necesario,
-- pero lo ideal es configurar políticas. Aquí las dejo abiertas para facilitar el desarrollo inicial según pedido.

ALTER TABLE gym_recipes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow all for gym_recipes" ON gym_recipes FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE gym_routines ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow all for gym_routines" ON gym_routines FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE gym_trainers ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow all for gym_trainers" ON gym_trainers FOR ALL USING (true) WITH CHECK (true);
