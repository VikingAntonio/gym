-- 1. Asegurar que no haya conflictos con la política si ya se ejecutó antes
DROP POLICY IF EXISTS "Admins have full access" ON public.gym_users;

-- 2. Crear la nueva política para permitir a los admins gestionar todo
-- Esta política permite a cualquier usuario con rol 'gym-admin' hacer TODO en gym_users
CREATE POLICY "Admins have full access" ON public.gym_users
FOR ALL
TO authenticated
USING ( (SELECT role FROM public.gym_users WHERE id = auth.uid()) = 'gym-admin' );

-- 3. Asignar el rol de Administrador Global al usuario solicitado
-- Usamos lower() y trim() para asegurar que encuentre el correo sin importar mayúsculas o espacios.
UPDATE public.gym_users
SET role = 'gym-admin'
WHERE lower(trim(email)) = lower(trim('designerantoniomontiel@gmail.com'));
