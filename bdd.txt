-- 0. PREPARACIÓN: Limpiar todo rastro de versiones anteriores para evitar conflictos
DO $$
DECLARE
    r RECORD;
BEGIN
    -- Eliminar Foreign Keys
    FOR r IN (
        SELECT table_name, constraint_name
        FROM information_schema.table_constraints
        WHERE table_name IN ('gym_users', 'gym_recipes', 'gym_routines', 'gym_trainers')
          AND constraint_type = 'FOREIGN KEY'
    ) LOOP
        EXECUTE 'ALTER TABLE ' || quote_ident(r.table_name) || ' DROP CONSTRAINT IF EXISTS ' || quote_ident(r.constraint_name);
    END LOOP;

    -- Eliminar Constraints de Unicidad que puedan bloquear el registro
    ALTER TABLE IF EXISTS gym_users DROP CONSTRAINT IF EXISTS gym_users_username_key;
    ALTER TABLE IF EXISTS gym_users DROP CONSTRAINT IF EXISTS gym_users_email_key;
    ALTER TABLE IF EXISTS gym_users DROP CONSTRAINT IF EXISTS gym_users_domain_key;
END $$;

-- 1. ASEGURAR TABLAS CON TIPOS CORRECTOS
CREATE TABLE IF NOT EXISTS gym_users (
    id UUID PRIMARY KEY,
    username TEXT,
    full_name TEXT,
    email TEXT,
    domain TEXT,
    role TEXT DEFAULT 'gym-owner',
    owner_id UUID,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS gym_recipes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID,
    title TEXT,
    description TEXT,
    ingredients TEXT,
    instructions TEXT,
    image_url TEXT,
    category TEXT,
    calories INTEGER,
    protein INTEGER,
    carbs INTEGER,
    fats INTEGER,
    prep_time TEXT DEFAULT '20 min',
    difficulty TEXT DEFAULT 'Media',
    diet_type TEXT DEFAULT 'Equilibrada',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS gym_routines (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID,
    title TEXT,
    description TEXT,
    exercises JSONB DEFAULT '[]'::jsonb,
    level TEXT,
    goal TEXT,
    image_url TEXT,
    target_gender TEXT DEFAULT 'Unisex',
    target_age_group TEXT DEFAULT 'Adulto',
    health_conditions TEXT DEFAULT 'Ninguna',
    plan_duration_weeks INTEGER DEFAULT 4,
    difficulty_level TEXT DEFAULT 'Intermedio',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS gym_trainers (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID,
    name TEXT,
    specialty TEXT,
    bio TEXT,
    image_url TEXT,
    contact_info TEXT,
    rating DECIMAL(3,2) DEFAULT 5.0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. MIGRACIÓN DE DATOS Y TIPOS (Rescate de BIGINT a UUID)
DO $$
BEGIN
    -- Limpieza de IDs antiguos en gym_users
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'gym_users' AND column_name = 'id' AND data_type != 'uuid') THEN
        DELETE FROM gym_users WHERE id::text !~ '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$';
        ALTER TABLE gym_users ALTER COLUMN id TYPE UUID USING id::text::uuid;
    END IF;

    -- Limpieza de owner_id en todas las tablas
    UPDATE gym_users SET owner_id = NULL WHERE owner_id::text !~ '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$';
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'gym_users' AND column_name = 'owner_id' AND data_type != 'uuid') THEN
        ALTER TABLE gym_users ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;
    END IF;

    UPDATE gym_recipes SET owner_id = NULL WHERE owner_id::text !~ '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$';
    ALTER TABLE gym_recipes ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;

    UPDATE gym_routines SET owner_id = NULL WHERE owner_id::text !~ '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$';
    ALTER TABLE gym_routines ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;

    UPDATE gym_trainers SET owner_id = NULL WHERE owner_id::text !~ '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$';
    ALTER TABLE gym_trainers ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;
END $$;

-- 5. RPC "SECURITY BRIDGE" (EL PUENTE DE PLATA)
-- Esta función tiene permisos elevados para crear el perfil ignorando temporalmente el RLS
CREATE OR REPLACE FUNCTION public.create_profile_if_missing(
    p_id UUID,
    p_email TEXT,
    p_full_name TEXT DEFAULT '',
    p_domain TEXT DEFAULT NULL,
    p_role TEXT DEFAULT 'gym-owner',
    p_owner_id UUID DEFAULT NULL
) RETURNS JSONB AS $$
DECLARE
    v_username TEXT;
    v_profile RECORD;
BEGIN
    SET search_path = public;

    -- 1. Limpieza total de cualquier registro previo que use ese ID o Email
    DELETE FROM public.gym_users WHERE id = p_id OR email = p_email;

    -- 2. Generar Username basado en email + fragmento de ID
    v_username := split_part(p_email, '@', 1) || '_' || substr(p_id::text, 1, 8);

    -- 3. Insertar con ON CONFLICT por si acaso
    INSERT INTO public.gym_users (id, email, username, full_name, domain, role, owner_id)
    VALUES (p_id, p_email, v_username, p_full_name, p_domain, p_role, p_owner_id)
    ON CONFLICT (id) DO UPDATE SET
        full_name = COALESCE(NULLIF(EXCLUDED.full_name, ''), public.gym_users.full_name),
        domain = COALESCE(NULLIF(EXCLUDED.domain, ''), public.gym_users.domain),
        owner_id = COALESCE(EXCLUDED.owner_id, public.gym_users.owner_id);

    -- 4. Devolver el perfil resultante
    SELECT * INTO v_profile FROM public.gym_users WHERE id = p_id;
    RETURN row_to_json(v_profile)::jsonb;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Permisos de ejecución para todos
GRANT EXECUTE ON FUNCTION public.create_profile_if_missing TO authenticated, anon, service_role;

-- 6. TRIGGER DE SINCRONIZACIÓN AUTOMÁTICA
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  SET search_path = public;
  PERFORM public.create_profile_if_missing(
    new.id,
    new.email,
    COALESCE(new.raw_user_meta_data->>'full_name', ''),
    NULLIF(new.raw_user_meta_data->>'domain', ''),
    COALESCE(new.raw_user_meta_data->>'role', 'gym-owner'),
    (NULLIF(new.raw_user_meta_data->>'owner_id', ''))::uuid
  );
  RETURN new;
EXCEPTION WHEN OTHERS THEN
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
