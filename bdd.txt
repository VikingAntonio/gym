-- Copia y pega estos comandos en el SQL Editor de Supabase.

-- Tabla de Recetas (Nutrición)
-- Almacena platos saludables con sus ingredientes e instrucciones.
CREATE TABLE IF NOT EXISTS gym_recipes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    ingredients TEXT, -- Puede ser texto simple o formateado
    instructions TEXT,
    image_url TEXT,
    category TEXT, -- Desayuno, Almuerzo, Cena, Snack, Postre
    calories INTEGER,
    protein INTEGER,
    carbs INTEGER,
    fats INTEGER,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Tabla de Rutinas
-- Almacena planes de entrenamiento con estructura semanal en JSONB.
CREATE TABLE IF NOT EXISTS gym_routines (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    exercises JSONB DEFAULT '[]'::jsonb, -- Estructura: [{"day": "Lunes", "moves": [{"name": "Sentadilla", "sets": 3, "reps": 12, "rest": "60s"}]}]
    level TEXT, -- Principiante, Intermedio, Avanzado
    goal TEXT, -- Perder peso, Ganar músculo, Tonificar, Resistencia
    image_url TEXT,
    duration_weeks INTEGER DEFAULT 4,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Tabla de Entrenadores
-- Perfiles de los profesionales del gimnasio.
CREATE TABLE IF NOT EXISTS gym_trainers (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    specialty TEXT,
    bio TEXT,
    image_url TEXT,
    contact_info TEXT, -- WhatsApp, Email, Instagram
    rating DECIMAL(3,2) DEFAULT 5.0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- ######################################################
-- # POLÍTICAS DE SEGURIDAD (RLS)                       #
-- ######################################################
-- IMPORTANTE: Estas políticas permiten acceso total (CRUD) para facilitar el desarrollo inicial.
-- En producción, se recomienda restringir INSERT/UPDATE/DELETE a usuarios autenticados.

ALTER TABLE gym_recipes ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow all for gym_recipes" ON gym_recipes;
CREATE POLICY "Allow all for gym_recipes" ON gym_recipes FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE gym_routines ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow all for gym_routines" ON gym_routines;
CREATE POLICY "Allow all for gym_routines" ON gym_routines FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE gym_trainers ENABLE ROW LEVEL SECURITY;
DROP POLICY IF EXISTS "Allow all for gym_trainers" ON gym_trainers;
CREATE POLICY "Allow all for gym_trainers" ON gym_trainers FOR ALL USING (true) WITH CHECK (true);
