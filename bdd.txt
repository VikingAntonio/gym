-- 1. Asegurar que la política de lectura sea pública para que todos vean la lista (necesario para fetchUsers)
DROP POLICY IF EXISTS "Public profiles are viewable by everyone" ON public.wisbe_users;
CREATE POLICY "Public profiles are viewable by everyone" ON public.wisbe_users FOR SELECT USING (true);

-- 2. Asegurar que no haya conflictos con la política de Admin si ya se ejecutó antes
DROP POLICY IF EXISTS "Admins have full access" ON public.wisbe_users;

-- 3. Crear la nueva política para permitir a los admins gestionar todo
-- Nota: Usamos public.is_admin() para evitar recursividad infinita en RLS
CREATE POLICY "Admins have full access" ON public.wisbe_users
FOR ALL
TO authenticated
USING ( public.is_admin() );

-- 4. Asignar el rol de Administrador Global al usuario solicitado
-- Usamos ILIKE y trim() para mayor flexibilidad en la búsqueda del correo.
UPDATE public.wisbe_users
SET role = 'gym-admin'
WHERE email ILIKE 'designerantoniomontiel@gmail.com';
