-- 0. MIGRACIONES PREVIAS (Asegurar tipos UUID para evitar error uuid = bigint)
DO $$
BEGIN
    -- Recetas
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'gym_recipes' AND column_name = 'owner_id' AND data_type != 'uuid') THEN
        ALTER TABLE gym_recipes ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;
    END IF;
    -- Rutinas
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'gym_routines' AND column_name = 'owner_id' AND data_type != 'uuid') THEN
        ALTER TABLE gym_routines ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;
    END IF;
    -- Entrenadores
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'gym_trainers' AND column_name = 'owner_id' AND data_type != 'uuid') THEN
        ALTER TABLE gym_trainers ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;
    END IF;
    -- Usuarios
    IF EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name = 'gym_users' AND column_name = 'owner_id' AND data_type != 'uuid') THEN
        ALTER TABLE gym_users ALTER COLUMN owner_id TYPE UUID USING owner_id::text::uuid;
    END IF;
END $$;

-- 1. TABLA DE USUARIOS (PERFILES)
CREATE TABLE IF NOT EXISTS gym_users (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    username TEXT UNIQUE NOT NULL,
    full_name TEXT,
    email TEXT UNIQUE NOT NULL,
    domain TEXT, -- OPCIONAL
    role TEXT CHECK (role IN ('gym-admin', 'gym-owner', 'gym-entrenador', 'gym-socios')) DEFAULT 'gym-owner',
    owner_id UUID REFERENCES gym_users(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Asegurar que el dominio no sea obligatorio y no tenga restricción única
ALTER TABLE gym_users ALTER COLUMN domain DROP NOT NULL;
ALTER TABLE gym_users DROP CONSTRAINT IF EXISTS gym_users_domain_key;

-- 2. TABLA DE RECETAS (NUTRICIÓN)
CREATE TABLE IF NOT EXISTS gym_recipes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID REFERENCES gym_users(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT,
    ingredients TEXT,
    instructions TEXT,
    image_url TEXT,
    category TEXT,
    calories INTEGER,
    protein INTEGER,
    carbs INTEGER,
    fats INTEGER,
    prep_time TEXT DEFAULT '20 min',
    difficulty TEXT DEFAULT 'Media',
    diet_type TEXT DEFAULT 'Equilibrada',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 3. TABLA DE RUTINAS
CREATE TABLE IF NOT EXISTS gym_routines (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID REFERENCES gym_users(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT,
    exercises JSONB DEFAULT '[]'::jsonb,
    level TEXT,
    goal TEXT,
    image_url TEXT,
    target_gender TEXT DEFAULT 'Unisex',
    target_age_group TEXT DEFAULT 'Adulto',
    health_conditions TEXT DEFAULT 'Ninguna',
    plan_duration_weeks INTEGER DEFAULT 4,
    difficulty_level TEXT DEFAULT 'Intermedio',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 4. TABLA DE ENTRENADORES
CREATE TABLE IF NOT EXISTS gym_trainers (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    owner_id UUID REFERENCES gym_users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    specialty TEXT,
    bio TEXT,
    image_url TEXT,
    contact_info TEXT,
    rating DECIMAL(3,2) DEFAULT 5.0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- POLÍTICAS PARA gym_users
DROP POLICY IF EXISTS "Public profiles are viewable by everyone" ON gym_users;
CREATE POLICY "Public profiles are viewable by everyone" ON gym_users FOR SELECT USING (true);

DROP POLICY IF EXISTS "Users can update own profile" ON gym_users;
CREATE POLICY "Users can update own profile" ON gym_users FOR UPDATE USING ((auth.uid())::uuid = id);

DROP POLICY IF EXISTS "Owners can manage their staff and socios" ON gym_users;
CREATE POLICY "Owners can manage their staff and socios" ON gym_users
FOR ALL USING ((auth.uid())::uuid = owner_id);

-- POLÍTICAS PARA gym_recipes
DROP POLICY IF EXISTS "Anyone can view recipes" ON gym_recipes;
CREATE POLICY "Anyone can view recipes" ON gym_recipes FOR SELECT USING (true);

DROP POLICY IF EXISTS "Owners and staff can manage recipes" ON gym_recipes;

    (auth.uid())::uuid = owner_id OR
    EXISTS (SELECT 1 FROM gym_users u WHERE u.id = (auth.uid())::uuid AND u.owner_id = gym_recipes.owner_id)
);

-- Función para manejar nuevos registros de Auth y crear el perfil automáticamente
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.gym_users (id, email, username, full_name, domain, role, owner_id)
  VALUES (
    new.id,
    new.email,
    COALESCE(new.raw_user_meta_data->>'username', split_part(new.email, '@', 1) || '_' || substr(new.id::text, 1, 4)),
    COALESCE(new.raw_user_meta_data->>'full_name', ''),
    NULLIF(new.raw_user_meta_data->>'domain', ''),
    COALESCE(new.raw_user_meta_data->>'role', 'gym-owner'),
    (NULLIF(new.raw_user_meta_data->>'owner_id', ''))::uuid
  );
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Trigger para ejecutar la función al crear un usuario en auth.users
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
