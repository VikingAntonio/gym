-- Ejecuta esto en el SQL Editor de Supabase si tienes problemas
-- para ver o crear tus recetas.

-- 1. ASEGURAR ESTRUCTURA DE TABLA
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS calories INTEGER DEFAULT 0;
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS protein INTEGER DEFAULT 0;
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS carbs INTEGER DEFAULT 0;
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS fats INTEGER DEFAULT 0;
ALTER TABLE gym_recipes ADD COLUMN IF NOT EXISTS owner_id UUID;

-- 2. HABILITAR SEGURIDAD (RLS)
ALTER TABLE gym_recipes ENABLE ROW LEVEL SECURITY;

-- 3. ELIMINAR POLÍTICAS ANTIGUAS PARA EVITAR CONFLICTOS
DROP POLICY IF EXISTS "Public viewable" ON gym_recipes;
DROP POLICY IF EXISTS "Anyone can view recipes" ON gym_recipes;
DROP POLICY IF EXISTS "Owners manage recipes" ON gym_recipes;
DROP POLICY IF EXISTS "Owners and staff can manage recipes" ON gym_recipes;

-- 4. CREAR NUEVAS POLÍTICAS LIMPIAS
-- Permitir que CUALQUIERA vea las recetas (necesario para nutricion.html)
CREATE POLICY "Public recipes access"
ON gym_recipes FOR SELECT
USING (true);

-- Permitir que los DUEÑOS gestionen sus propias recetas
CREATE POLICY "Owners manage own recipes"
ON gym_recipes FOR ALL
USING (auth.uid() = owner_id);

-- 5. PERMISOS TOTALES PARA EVITAR BLOQUEOS DE ROL
GRANT ALL ON gym_recipes TO anon, authenticated, service_role;
GRANT ALL ON ALL SEQUENCES IN SCHEMA public TO anon, authenticated, service_role;
