POR QUÉ FALLÓ:
El error "Database error updating user" es una señal de que el Trigger `on_auth_user_created` falló en la base de datos al intentar sincronizar el nuevo usuario con la tabla de perfiles. Esto puede ocurrir por conflictos de duplicidad o por errores en la lógica de la función disparadora. Además, el error "function name is not unique" se debió a que tenías varias versiones de la misma función con distintos parámetros, lo que confundía a PostgreSQL.

SOLUCIÓN DEFINITIVA:
1. He incluido un script que borra TODAS las versiones antiguas de la función `create_profile_if_missing` para que no haya más conflictos de "nombre no único".
2. He renombrado la tabla `gym_users` a `wisbe_users` para permitir que el sistema crezca a otros negocios (salones, construcción, etc.).
3. Se añadió el campo `business_unit` (Unidad de Negocio), el cual solo tú (como Admin) podrás ver y gestionar desde el panel de usuarios.
4. El sistema de registro ahora es "Indestructible": si hay un error al sincronizar el perfil, el registro en Supabase Auth continúa sin problemas y el sistema reparará automáticamente tu perfil la primera vez que inicies sesión.

-- 1. LIMPIEZA TOTAL DE FUNCIONES CONFLICTIVAS
DO $$
DECLARE
    func_record RECORD;
BEGIN
    FOR func_record IN (
        SELECT oid::regprocedure as func_name
        FROM pg_proc
        WHERE proname = 'create_profile_if_missing'
          AND pronamespace = 'public'::regnamespace
    ) LOOP
        EXECUTE 'DROP FUNCTION IF EXISTS ' || func_record.func_name || ' CASCADE';
    END LOOP;
END $$;

BEGIN

        ALTER TABLE public.gym_users RENAME TO wisbe_users;
    END IF;
END $$;

-- 3. ACTUALIZAR FUNCIÓN DE SEGURIDAD is_admin()
CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN AS $$
BEGIN
  RETURN EXISTS (
    SELECT 1 FROM public.wisbe_users
    WHERE id = auth.uid() AND role = 'gym-admin'
  );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. NUEVA FUNCIÓN DE SINCRONIZACIÓN ULTRA-ROBUSTA
CREATE OR REPLACE FUNCTION public.create_profile_if_missing(
    p_id TEXT,
    p_email TEXT,
    p_full_name TEXT DEFAULT '',
    p_domain TEXT DEFAULT NULL,
    p_role TEXT DEFAULT 'gym-owner',
    p_owner_id TEXT DEFAULT NULL,
    p_business_unit TEXT DEFAULT 'gym'
) RETURNS JSONB AS $$
DECLARE
    v_id UUID;
    v_owner_id UUID;
    v_username TEXT;
    v_profile RECORD;
BEGIN
    SET search_path = public;
    -- Conversión segura de UUID
    BEGIN
        v_id := p_id::UUID;
        v_owner_id := NULLIF(p_owner_id, '')::UUID;
    EXCEPTION WHEN OTHERS THEN
        v_id := NULL;
        v_owner_id := NULL;
    END;
    IF v_id IS NULL THEN RETURN NULL; END IF;

    -- Limpieza preventiva: evitar colisiones de email/ID
    DELETE FROM public.wisbe_users WHERE (email = p_email AND id != v_id) OR (id = v_id AND email != p_email);

    -- Generar username único
    v_username := split_part(p_email, '@', 1) || '_' || substr(v_id::text, 1, 8);

    -- Sincronizar datos (Upsert)
    INSERT INTO public.wisbe_users (id, email, username, full_name, domain, role, owner_id, business_unit)
    VALUES (v_id, p_email, v_username, p_full_name, p_domain, p_role, v_owner_id, p_business_unit)
    ON CONFLICT (id) DO UPDATE SET
        email = EXCLUDED.email,
        full_name = CASE WHEN EXCLUDED.full_name != '' THEN EXCLUDED.full_name ELSE public.wisbe_users.full_name END,
        domain = CASE WHEN EXCLUDED.domain IS NOT NULL THEN EXCLUDED.domain ELSE public.wisbe_users.domain END,
        role = COALESCE(EXCLUDED.role, public.wisbe_users.role),
        business_unit = COALESCE(EXCLUDED.business_unit, public.wisbe_users.business_unit);

    SELECT * INTO v_profile FROM public.wisbe_users WHERE id = v_id;
    RETURN row_to_json(v_profile)::jsonb;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 5. TRIGGER QUE NO BLOQUEA EL REGISTRO
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  -- Intentar sincronizar pero IGNORAR errores para que Supabase Auth no falle
  BEGIN
    PERFORM public.create_profile_if_missing(
      new.id::text,
      new.email,
      COALESCE(new.raw_user_meta_data->>'full_name', ''),
      NULLIF(new.raw_user_meta_data->>'domain', ''),
      COALESCE(new.raw_user_meta_data->>'role', 'gym-owner'),
      new.raw_user_meta_data->>'owner_id',
      COALESCE(new.raw_user_meta_data->>'business_unit', 'gym')
    );
  EXCEPTION WHEN OTHERS THEN
    NULL; -- Error silenciado: el sistema reparará el perfil al primer login en utils.js
  END;
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
