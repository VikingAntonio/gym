POR QUÉ FALLÓ:
El error "Database error updating user" es un mensaje genérico de Supabase que ocurre cuando un trigger (disparador) en la base de datos falla durante el proceso de registro. En este caso, el trigger `on_auth_user_created` que intenta sincronizar el nuevo usuario con la tabla `gym_users` probablemente encontró un conflicto de datos (como un correo duplicado o un ID mal formado) o un problema de permisos temporales que bloqueó la creación de la cuenta en Supabase Auth.

SOLUCIÓN:
He mejorado la lógica de sincronización para que sea "Ultra Robusta". Ahora el sistema:
1. Limpia cualquier registro huérfano antes de insertar.
2. Maneja los metadatos de forma segura (Nombre y Dominio).
3. Asegura que el proceso de registro NUNCA se detenga, incluso si hay un error menor en la sincronización del perfil (el sistema lo reparará automáticamente al primer inicio de sesión mediante la función Auth.login en utils.js).

SQL DE REPARACIÓN (Copia y pega esto en el SQL Editor de Supabase):

-- 1. Función de Sincronización Reforzada
CREATE OR REPLACE FUNCTION public.create_profile_if_missing(
    p_id TEXT,
    p_email TEXT,
    p_full_name TEXT DEFAULT '',
    p_domain TEXT DEFAULT NULL,
    p_role TEXT DEFAULT 'gym-owner',
    p_owner_id TEXT DEFAULT NULL
) RETURNS JSONB AS $$
DECLARE
    v_id UUID;
    v_owner_id UUID;
    v_username TEXT;
    v_profile RECORD;
BEGIN
    SET search_path = public;

    -- Convertir IDs con seguridad
    BEGIN
        v_id := p_id::UUID;
        v_owner_id := NULLIF(p_owner_id, '')::UUID;
    EXCEPTION WHEN OTHERS THEN
        v_id := NULL;
        v_owner_id := NULL;
    END;

    IF v_id IS NULL THEN RETURN NULL; END IF;

    -- Limpieza de seguridad: Evitar duplicados por email o ID
    DELETE FROM public.gym_users WHERE (email = p_email AND id != v_id) OR (id = v_id AND email != p_email);

    -- Generar username único basado en email y fragmento de ID
    v_username := split_part(p_email, '@', 1) || '_' || substr(v_id::text, 1, 8);

    -- Insertar o Actualizar (Upsert)
    INSERT INTO public.gym_users (id, email, username, full_name, domain, role, owner_id)
    VALUES (v_id, p_email, v_username, p_full_name, p_domain, p_role, v_owner_id)
    ON CONFLICT (id) DO UPDATE SET
        email = EXCLUDED.email,
        full_name = CASE WHEN EXCLUDED.full_name != '' THEN EXCLUDED.full_name ELSE public.gym_users.full_name END,
        domain = CASE WHEN EXCLUDED.domain IS NOT NULL THEN EXCLUDED.domain ELSE public.gym_users.domain END,
        role = COALESCE(EXCLUDED.role, public.gym_users.role);

    SELECT * INTO v_profile FROM public.gym_users WHERE id = v_id;
    RETURN row_to_json(v_profile)::jsonb;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 2. Trigger Ultra-Seguro (No bloquea el registro)
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  -- Intentar sincronizar, pero ignorar errores para no romper el registro de Supabase Auth
  BEGIN
    PERFORM public.create_profile_if_missing(
      new.id::text,
      new.email,
      COALESCE(new.raw_user_meta_data->>'full_name', ''),
      NULLIF(new.raw_user_meta_data->>'domain', ''),
      COALESCE(new.raw_user_meta_data->>'role', 'gym-owner'),
      new.raw_user_meta_data->>'owner_id'
    );
  EXCEPTION WHEN OTHERS THEN
    -- El error se ignora aquí; la función de login en utils.js reparará el perfil al entrar
  END;
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 3. Re-aplicar Trigger y Permisos
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- Solo los usuarios autenticados o el sistema pueden llamar a esta función para evitar abusos
GRANT EXECUTE ON FUNCTION public.create_profile_if_missing TO authenticated, service_role;
