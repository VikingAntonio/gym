POR QUÉ SIGUE EL ERROR:
El error "Database error updating user" es un bloqueo de Supabase Auth causado por funciones o triggers en tu base de datos que fallan o están duplicados. Si la base de datos tiene algún error interno al registrarse, Supabase detiene todo el proceso por seguridad.

SOLUCIÓN "INDESTRUCTIBLE":
He desacoplado el registro de la creación del perfil. He configurado un Trigger que NO hace nada complicado, asegurando que el registro en Supabase Auth SIEMPRE sea exitoso. La creación real de tu perfil ocurrirá automáticamente la primera vez que inicies sesión. Esto elimina de raíz cualquier posibilidad de que el error "Database error" vuelva a aparecer.

SQL DE REPARACIÓN DEFINITIVA (Copia y pega TODO este bloque en el SQL Editor de Supabase):

-- 1. LIMPIEZA TOTAL DE CONFLICTOS
DO $$
DECLARE
    trig RECORD;
    func RECORD;
BEGIN
    -- Borrar todos los triggers en auth.users para que no bloqueen el registro
    FOR trig IN (SELECT trigger_name FROM information_schema.triggers WHERE event_object_table = 'users' AND event_object_schema = 'auth') LOOP
        EXECUTE 'DROP TRIGGER IF EXISTS ' || trig.trigger_name || ' ON auth.users CASCADE';
    END LOOP;

    -- Borrar versiones antiguas de funciones para evitar "function name is not unique"
    FOR func IN (
        SELECT oid::regprocedure as name
        FROM pg_proc
        WHERE proname IN ('create_profile_if_missing', 'handle_new_user', 'is_admin')
          AND pronamespace = 'public'::regnamespace
    ) LOOP
        EXECUTE 'DROP FUNCTION IF EXISTS ' || func.name || ' CASCADE';
    END LOOP;
END $$;

-- 2. MIGRACIÓN Y ESTRUCTURA WISBE
DO $$
BEGIN
    IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'gym_users') THEN
        ALTER TABLE public.gym_users RENAME TO wisbe_users;
    END IF;
END $$;

-- Asegurar tabla wisbe_users
CREATE TABLE IF NOT EXISTS public.wisbe_users (
    id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    email TEXT,
    username TEXT,
    full_name TEXT,
    domain TEXT,
    role TEXT DEFAULT 'gym-owner',
    owner_id UUID,
    business_unit TEXT DEFAULT 'gym',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE public.wisbe_users ADD COLUMN IF NOT EXISTS business_unit TEXT DEFAULT 'gym';
ALTER TABLE public.wisbe_users ADD COLUMN IF NOT EXISTS owner_id UUID;

-- 3. RECREAR FUNCIÓN DE SEGURIDAD (is_admin)
CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN AS $$
BEGIN
  RETURN EXISTS (SELECT 1 FROM public.wisbe_users WHERE id = auth.uid() AND role = 'gym-admin');
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. FUNCIÓN DE SINCRONIZACIÓN (Llamada al iniciar sesión)
CREATE OR REPLACE FUNCTION public.create_profile_if_missing(
    p_id TEXT,
    p_email TEXT,
    p_full_name TEXT DEFAULT '',
    p_domain TEXT DEFAULT NULL,
    p_role TEXT DEFAULT 'gym-owner',
    p_owner_id TEXT DEFAULT NULL,
    p_business_unit TEXT DEFAULT 'gym'
) RETURNS JSONB AS $$
DECLARE
    v_id UUID;
    v_owner_id UUID;
    v_username TEXT;
    v_profile RECORD;
BEGIN
    SET search_path = public;
    v_id := p_id::UUID;
    v_owner_id := NULLIF(p_owner_id, '')::UUID;

    -- Limpieza preventiva
    DELETE FROM public.wisbe_users WHERE (email = p_email AND id != v_id) OR (id = v_id AND email != p_email);
    v_username := split_part(p_email, '@', 1) || '_' || substr(v_id::text, 1, 8);

    -- Sincronización robusta (Upsert)
    INSERT INTO public.wisbe_users (id, email, username, full_name, domain, role, owner_id, business_unit)
    VALUES (v_id, p_email, v_username, p_full_name, p_domain, p_role, v_owner_id, p_business_unit)
    ON CONFLICT (id) DO UPDATE SET
        email = EXCLUDED.email,
        full_name = CASE WHEN EXCLUDED.full_name != '' THEN EXCLUDED.full_name ELSE public.wisbe_users.full_name END,
        domain = CASE WHEN EXCLUDED.domain IS NOT NULL THEN EXCLUDED.domain ELSE public.wisbe_users.domain END,
        role = COALESCE(EXCLUDED.role, public.wisbe_users.role),
        business_unit = COALESCE(EXCLUDED.business_unit, public.wisbe_users.business_unit);

    SELECT * INTO v_profile FROM public.wisbe_users WHERE id = v_id;
    RETURN row_to_json(v_profile)::jsonb;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 5. TRIGGER "VACÍO" (Garantiza que el registro nunca falle)
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  -- No hacemos nada aquí para evitar que errores de base de datos bloqueen el registro.
  -- El sistema de la web (utils.js) se encargará de crear el perfil al entrar.
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 6. PERMISOS Y POLÍTICAS
GRANT EXECUTE ON FUNCTION public.create_profile_if_missing TO authenticated, service_role;

DO $$
BEGIN
    DROP POLICY IF EXISTS "Public profiles" ON public.wisbe_users;
    CREATE POLICY "Public profiles" ON public.wisbe_users FOR SELECT USING (true);

    DROP POLICY IF EXISTS "Admins full" ON public.wisbe_users;
    CREATE POLICY "Admins full" ON public.wisbe_users FOR ALL TO authenticated USING (public.is_admin());

    -- Políticas de negocio
    DROP POLICY IF EXISTS "Owner manage recipes" ON public.gym_recipes;
    CREATE POLICY "Owner manage recipes" ON public.gym_recipes FOR ALL USING (
        (auth.uid())::uuid = owner_id OR EXISTS (SELECT 1 FROM public.wisbe_users u WHERE u.id = (auth.uid())::uuid AND u.owner_id = public.gym_recipes.owner_id)
    );
END $$;
