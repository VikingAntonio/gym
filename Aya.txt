El error "Database error updating user" ocurrió por un fallo en el trigger de sincronización al intentar crear el perfil en la tabla de usuarios. He implementado una versión "Ultra Robusta" que no bloquea el registro y se auto-repara.

NUEVA ACTUALIZACIÓN: CAMBIO DE NOMBRE DE TABLA Y MULTI-NEGOCIO
Siguiendo tus instrucciones, la tabla `gym_users` ha sido renombrada a `wisbe_users` para permitir que el sistema sea escalable a otros negocios (salones, construcción, etc.). Se ha añadido el campo `business_unit` para diferenciar estos registros, el cual solo es visible para Administradores Globales.

SQL DE MIGRACIÓN Y REPARACIÓN (Copia y pega todo esto en Supabase):

-- 1. Renombrar la tabla principal y añadir columna de unidad de negocio
DO $$
BEGIN
    IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'public' AND table_name = 'gym_users') THEN
        ALTER TABLE public.gym_users RENAME TO wisbe_users;
    END IF;
END $$;
ALTER TABLE public.wisbe_users ADD COLUMN IF NOT EXISTS business_unit TEXT DEFAULT 'gym';

-- 2. Actualizar Función de Seguridad is_admin()
CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN AS $$
BEGIN
  RETURN EXISTS (
    SELECT 1
    FROM public.wisbe_users
    WHERE id = auth.uid() AND role = 'gym-admin'
  );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 3. Función de Sincronización Reforzada (wisbe_users)
CREATE OR REPLACE FUNCTION public.create_profile_if_missing(
    p_id TEXT,
    p_email TEXT,
    p_full_name TEXT DEFAULT '',
    p_domain TEXT DEFAULT NULL,
    p_role TEXT DEFAULT 'gym-owner',
    p_owner_id TEXT DEFAULT NULL,
    p_business_unit TEXT DEFAULT 'gym'
) RETURNS JSONB AS $$
DECLARE
    v_id UUID;
    v_owner_id UUID;
    v_username TEXT;
    v_profile RECORD;
BEGIN
    SET search_path = public;

    BEGIN
        v_id := p_id::UUID;
        v_owner_id := NULLIF(p_owner_id, '')::UUID;
    EXCEPTION WHEN OTHERS THEN
        v_id := NULL;
        v_owner_id := NULL;
    END;

    IF v_id IS NULL THEN RETURN NULL; END IF;

    DELETE FROM public.wisbe_users WHERE (email = p_email AND id != v_id) OR (id = v_id AND email != p_email);

    v_username := split_part(p_email, '@', 1) || '_' || substr(v_id::text, 1, 8);

    INSERT INTO public.wisbe_users (id, email, username, full_name, domain, role, owner_id, business_unit)
    VALUES (v_id, p_email, v_username, p_full_name, p_domain, p_role, v_owner_id, p_business_unit)
    ON CONFLICT (id) DO UPDATE SET
        email = EXCLUDED.email,
        full_name = CASE WHEN EXCLUDED.full_name != '' THEN EXCLUDED.full_name ELSE public.wisbe_users.full_name END,
        domain = CASE WHEN EXCLUDED.domain IS NOT NULL THEN EXCLUDED.domain ELSE public.wisbe_users.domain END,
        role = COALESCE(EXCLUDED.role, public.wisbe_users.role),
        business_unit = COALESCE(EXCLUDED.business_unit, public.wisbe_users.business_unit);

    SELECT * INTO v_profile FROM public.wisbe_users WHERE id = v_id;
    RETURN row_to_json(v_profile)::jsonb;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. Trigger Ultra-Seguro actualizado
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  BEGIN
    PERFORM public.create_profile_if_missing(
      new.id::text,
      new.email,
      COALESCE(new.raw_user_meta_data->>'full_name', ''),
      NULLIF(new.raw_user_meta_data->>'domain', ''),
      COALESCE(new.raw_user_meta_data->>'role', 'gym-owner'),
      new.raw_user_meta_data->>'owner_id',
      COALESCE(new.raw_user_meta_data->>'business_unit', 'gym')
    );
  EXCEPTION WHEN OTHERS THEN
    NULL;
  END;
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 5. Re-vincular Trigger y Políticas (Supabase suele mantenerlas al renombrar, pero aseguramos)
DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- Actualizar políticas de otras tablas que referenciaban wisbe_users
DO $$
DECLARE
    pol RECORD;
BEGIN
    -- Borrar y recrear políticas que usan subconsultas a la tabla de usuarios
    DROP POLICY IF EXISTS "Owners and staff can manage recipes" ON public.gym_recipes;
    CREATE POLICY "Owners and staff can manage recipes" ON public.gym_recipes FOR ALL USING (

    );
