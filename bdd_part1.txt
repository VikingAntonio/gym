-- 0. PREPARACIÓN TOTAL: Limpiar todo lo que pueda causar errores de "ya existe"
DO $$
DECLARE
    pol RECORD;
    con RECORD;
BEGIN
    -- 1. Eliminar todas las políticas existentes (gym_*)
    FOR pol IN (SELECT policyname, tablename FROM pg_policies WHERE schemaname = 'public' AND tablename LIKE 'gym_%') LOOP
        EXECUTE format('DROP POLICY IF EXISTS %I ON %I', pol.policyname, pol.tablename);
    END LOOP;

    -- 2. Eliminar TODOS los constraints existentes (PK, FK, Check, Unique) e índices de gym_*
    -- Esto garantiza que podamos alterar tipos de columnas y evitar errores de "ya existe" o índices duplicados
    FOR con IN (
        SELECT table_name, constraint_name
        FROM information_schema.table_constraints
        WHERE table_schema = 'public' AND table_name LIKE 'gym_%'
          AND constraint_type IN ('PRIMARY KEY', 'FOREIGN KEY', 'CHECK', 'UNIQUE')
    ) LOOP
        EXECUTE format('ALTER TABLE %I DROP CONSTRAINT IF EXISTS %I CASCADE', con.table_name, con.constraint_name);
    END LOOP;

    -- Eliminar índices (excluyendo los de PK que ya borramos arriba)
    FOR con IN (
        SELECT indexname FROM pg_indexes WHERE schemaname = 'public' AND tablename LIKE 'gym_%'
    ) LOOP
        EXECUTE format('DROP INDEX IF EXISTS %I CASCADE', con.indexname);
    END LOOP;

    -- 3. Limpieza manual de triggers y funciones viejas
    DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
    DROP FUNCTION IF EXISTS public.handle_new_user();
    DROP FUNCTION IF EXISTS public.create_profile_if_missing(UUID, TEXT, TEXT, TEXT, TEXT, UUID);
    DROP FUNCTION IF EXISTS public.create_profile_if_missing(TEXT, TEXT, TEXT, TEXT, TEXT, TEXT);
END $$;
