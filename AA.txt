-- 1. Función para verificar si un usuario es admin sin recursividad (SECURITY DEFINER)
CREATE OR REPLACE FUNCTION public.is_admin()
RETURNS BOOLEAN AS $$
BEGIN
  RETURN EXISTS (
    SELECT 1
    FROM public.gym_users
    WHERE id = auth.uid() AND role = 'gym-admin'
  );
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 3. Actualizar políticas de gym_users (PROTECCIÓN DE ESCALACIÓN DE PRIVILEGIOS)
DO $$
DECLARE
    pol RECORD;
BEGIN
    FOR pol IN (SELECT policyname FROM pg_policies WHERE schemaname = 'public' AND tablename = 'gym_users') LOOP
        EXECUTE format('DROP POLICY IF EXISTS %I ON public.gym_users', pol.policyname);
    END LOOP;
END $$;

-- LECTURA: Solo autenticados. Admins ven todo, Owners ven staff/sí mismos, Socios ven sí mismos.
CREATE POLICY "Users can view their own profile" ON public.gym_users FOR SELECT TO authenticated USING (auth.uid() = id);
CREATE POLICY "Owners can view their staff" ON public.gym_users FOR SELECT TO authenticated USING (auth.uid() = owner_id);
CREATE POLICY "Admins can view everyone" ON public.gym_users FOR SELECT TO authenticated USING (public.is_admin());

-- ESCRITURA: Solo Admins pueden Insertar, Actualizar o Borrar (Satisface "Owner no puede editar solo ver")
-- Esto también previene que Socios/Owners se cambien el rol a sí mismos (Escalación de privilegios).
CREATE POLICY "Only admins can manage users" ON public.gym_users FOR ALL TO authenticated USING (public.is_admin());

-- 4. Actualizar políticas de gym_recipes

BEGIN

        EXECUTE format('DROP POLICY IF EXISTS %I ON public.gym_recipes', pol.policyname);

CREATE POLICY "Anyone can view recipes" ON public.gym_recipes FOR SELECT USING (true);
CREATE POLICY "Admins can manage all recipes" ON public.gym_recipes FOR ALL TO authenticated USING (public.is_admin());
CREATE POLICY "Owners and staff can manage recipes" ON public.gym_recipes FOR ALL USING (
    (auth.uid())::uuid = owner_id OR
    EXISTS (SELECT 1 FROM public.gym_users u WHERE u.id = (auth.uid())::uuid AND u.owner_id = public.gym_recipes.owner_id)
);
